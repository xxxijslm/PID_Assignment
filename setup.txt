CREATE TABLE users
(userId int AUTO_INCREMENT NOT null PRIMARY KEY,
userName varchar(20) NOT null,
email varchar(40) NOT null,
`password` varchar(100) NOT null,
address varchar(40) DEFAULT '',
cartsId int not null)

CREATE TABLE categories
(categoryId int AUTO_INCREMENT NOT null PRIMARY KEY,
categoryName varchar(20) NOT null)

CREATE TABLE products
(productid int AUTO_INCREMENT NOT null PRIMARY KEY,
productName varchar(40) NOT null,
categoryId int NOT null,
price decimal(10,2) NOT NULL DEFAULT '0.00',
stock smallint NOT NULL DEFAULT '0',
decription varchar(100) DEFAULT '',
productImg varchar(20) NOT null)

CREATE TABLE orders
(orderId int(5) unsigned zerofill NOT NULL AUTO_INCREMENT PRIMARY KEY,
userId int NOT null,
orderDate datetime not null,
shippedDate datetime not null,
orderStatusId INT(11) NOT NULL DEFAULT '1')

CREATE TABLE orderDetails
(orderId int(5) unsigned zerofill NOT NULL,
productId int not null,
quantity smallint NOT NULL DEFAULT '0',
price decimal(10,2) NOT NULL DEFAULT '0.00')

CREATE TABLE orderStatus
(orderStatusId int NOT null AUTO_INCREMENT PRIMARY KEY,
orderStatusName varchar(20))


CREATE TABLE cartDetails
(userId int not null,
productId int not null,
price decimal(10,2) NOT NULL DEFAULT '0.00',
quantity smallint NOT NULL DEFAULT '0')
